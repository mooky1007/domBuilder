Create and nest components

        
    const H1 = Dom.h1.set({text: 'Hello World'});
    Dom.body.append(H1);

    // -------------------------------------------------

    const DIV = Dom.div;
    const H1 = Dom.h1;
    const SPAN = Dom.span;
    
    DIV.set({ class: "title" })
    H1.set({ text: 'Hello ' });
    SPAN.set({ style: { color: 'red' }, text: 'World' })

    DIV.set({
      children: [
        H1.set({
          children: [
            SPAN
          ]
        })
      ]
    })

    Dom.body.append(H1);

Adding inline-styles & class and id

        
      const Box = Dom.div.set({
        id: 'first_box',
        class: 'my_box',
        style: {
          width: '100px',
          height: '100px',
          border: '1px solid #000'
        }
      })

      Dom.body.append(Box)

You can also inject styles directly into the head tag.

          
      Dom.injectCss(`
        * {
          margin: 0;
          padding: 0;
          box-sizing: border-box;
        }

        .box {
          width: 100px;
          height: 100px;
          border: 1px solid #000;
        }
      `)

Event Binding & State And Effect

          
      const Counter = () => {
        const CounterButton = Dom.button;
        const [count, setCount] = Dom.state(0);
        Dom.effect(() => {
          CounterButton.text(`counted ${count()}`);
        }, [count])

        return CounterButton.set({
          on: {
            click: () => setCount(count() + 1)
          }
        });
      }

      Dom.body.append([
        Counter(),
        Counter(),
        Counter()
      ]);
          
        
          
      const CountDown = () => {
        const CountDown = Dom.span;
        const [count, setCount] = Dom.state(0);
        Dom.effect(() => {
          CountDown.text(`counted ${count()}`);
        }, [count])

        setInterval(() => setCount(count() + 1), 1000)

        return CountDown;
      }

      Dom.body.append(CountDown());
